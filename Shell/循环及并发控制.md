1. for循环

   for  变量名 [in 取值列表]

   do

   ​		循环体

   done

   1. 示例1：批量主机ping测试

      ```shell
      #!/bin/bash
      #ping check
      
      >ip.txt
      
      for i in {2..254}
      #for ip in `cat ip.txt`    #ip地址列表从ip.txt文件中读取
      do
              {
              ip=192.168.1.$i
              ping -c1 -w1 $ip &>/dev/null  #-w1  指定超时时间
              if [ $? -eq 0 ];then
                      echo "$ip is up"|tee -a ip.txt
              fi
              }&  #放到后台执行
      done
      wait   #等待之前所有的后台进程结束
      echo "finished..
      ```
